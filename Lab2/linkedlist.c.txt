}

void printList(struct linkedlist * head){
struct node *ptr = head->first;
printf("\n[ ");

while(ptr != NULL){
printf("%d, ", ptr->element);
ptr = ptr->next;
}

printf(" ]");
}

int search (struct linkedlist * head, int ele)
{
struct node*ptr=head->first;
if(ptr==NULL)
return 0;
while(ptr!=NULL)	
{
if(ptr->element==ele)
return 1;
ptr=ptr->next;
}
return 0;

}
